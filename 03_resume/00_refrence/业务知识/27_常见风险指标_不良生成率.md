# 不良生成率

## 1 业务含义

***不良生成率：***反映了纯新增不良金额（T+3月）【分子】占日均在贷正常M0余额【分母】的比例变化情况；

➤ **【分子】T+3月纯新增不良金额：**观察同笔借据从T+2月月末非不良状态转变成T+3月月末不良状态的不良余额【不良新发生】，即，五级分类状态从（'01','02'）变为（'03','04','05'）的借据余额；➤ **【分母】T月日均在贷正常M0余额；**

## 2 不良生成率

***Step1：计算T月的不良新发生余额【借据粒度：T-1月状态为非不良('01','02')，T月状态为不良('03','04','05')】;其中，考虑到借新还旧的旧借据在T-1月月末时点下可能为非不良，但在T月月末时点前结清产生的新借据可能为不良的情况，因此仍需进行借新还旧重组后以旧借据进关联分析不良新发生的情况；具体可参考风险成本率-核销前新增不良余额-借据维度的加工逻辑。***

***Step2：计算T月在贷正常M0贷款余额；***

***Step3：计算T月不良生成率：T+3月的不良新发生与T月的月日均贷款余额错位关联计算；***

## 3 参考代码-自然月口径-不良生成率

```SQL
-- step1 计算不良新发生-需进行借新还旧
with tmp as(
    select 
        t.*,
        row_number() over(partition by coalesce(a.origin_duebill_serialno,t.duebill_serialno),ds order by putout_date asc) as rn_asc,
        row_number() over(partition by coalesce(a.origin_duebill_serialno,t.duebill_serialno),ds order by putout_date desc) as rn_desc,
        coalesce(a.origin_duebill_serialno,t.duebill_serialno) as origin_duebill_serialno_traceback --借新还旧还原成老借据
    from brm_ent_dm_work.dwd_mcfcm_business_duebill_etl_ds t 
    left join( --借新还旧重组
        select t1.new_duebill_serialno_jxhj,
        coalesce(t5.old_duebill_serialno_jxhj,t4.old_duebill_serialno_jxhj,
                t3.old_duebill_serialno_jxhj,t2.old_duebill_serialno_jxhj,
                t1.old_duebill_serialno_jxhj) as origin_duebill_serialno
        from(select * from brm_ent_dm_work.dwd_jxhj_by_old_duebill where ds = '${run_date}')t1
        left join(select * from brm_ent_dm_work.dwd_jxhj_by_old_duebill where ds = '${run_date}')t2 on t1.old_duebill_serialno_jxhj = t2.new_duebill_serialno_jxhj
        left join(select * from brm_ent_dm_work.dwd_jxhj_by_old_duebill where ds = '${run_date}')t3 on t2.old_duebill_serialno_jxhj = t3.new_duebill_serialno_jxhj
        left join(select * from brm_ent_dm_work.dwd_jxhj_by_old_duebill where ds = '${run_date}')t4 on t3.old_duebill_serialno_jxhj = t4.new_duebill_serialno_jxhj
        left join(select * from brm_ent_dm_work.dwd_jxhj_by_old_duebill where ds = '${run_date}')t5 on t4.old_duebill_serialno_jxhj = t5.new_duebill_serialno_jxhj
    )a on t.duebill_serialno = a.new_duebill_serialno_jxhj
    where t.duestatus in ('0','1','2','3') and t.ds > '20230601' and substr(date_add(from_unixtime(unix_timestamp(ds,'yyyyMMdd'),'yyyy-MM-dd'),1),9,2) = '01' --月末
),temp as(
    select * from(
        select t.duebill_serialno as duebill_serialno_origin,a.duebill_serialno as duebill_serialno_lastest, --最新借据号 
            a.classifyresult,a.balance,t.ds,a.origin_duebill_serialno_traceback,row_number() over(partition by a.duebill_serialno,t.ds order by t.duebill_serialno desc) as rn
            from (select * from tmp where rn_asc = 1) t
            inner join(
                    select * from tmp where rn_desc = 1
                )a on a.origin_duebill_serialno_traceback = t.origin_duebill_serialno_traceback and t.ds = a.ds
    )
    where rn = 1
),temp1 as( --不良资产转让的五级分类
    select
        t.*,case when t0.duebill_serialno is not null and ds >= t0.package_date and ds < transfer_date then package_amt
                when t0.duebill_serialno is not null and ds >= t0.transfer_date then abs(package_amt - transfer_discount_amt)
                else t.balance end as  balance_lastest
    from temp t
    left join (
                select duebill_serialno,package_amt,transfer_discount_amt,replace(substr(package_date,1,10),'-','') as package_date,replace(substr(transfer_date,1,10),'-','') as transfer_date
                from brm_ent_dm_work.dwd_duebill_blzczr_ds 
                where ds ='${run_date}'
    ) t0 on t.duebill_serialno_lastest = t0.duebill_serialno 
)
select t1.ds_month,buliang_new_add/avg_balance as buliang_rate
from(
    select t.ds,sum(case when t_m.classifyresult in ('01','02') and t.classifyresult in ('03','04','05') then t.balance_lastest else 0 end) as buliang_new_add
    from temp1 t_m
    left join temp1 t on t.duebill_serialno_origin = t_m.duebill_serialno_origin and substr(t.ds,1,6) = replace(substr(date_add(from_unixtime(unix_timestamp(t_m.ds,'yyyyMMdd'),'yyyy-MM-dd'),1),1,7),'-','')
    group by t.ds
)t 
left join( --Step2 日均在贷M0余额 avg_balance 
    select substr(ds,1,6) as ds_month,
        sum(case when duestatus in ('0') and overduedays = 0 and coalesce(is_lyhg,0) = 1 then 0 
                 when duestatus in ('0') and overduedays = 0 and coalesce(is_bxdc,0) = 1 then we_balance_bxdc
                 when duestatus in ('0') and overduedays = 0 and coalesce(is_dbdc,0) = 1 then we_balance_dbdc
                 when duestatus in ('0') and overduedays = 0 then balance else 0 end)/count(distinct ds) as avg_balance
    from brm_ent_dm_work.dwd_mcfcm_business_duebill_etl_ds
    where ds >= '20230301'
    group by substr(ds,1,6)
)t1 on replace(substr(add_months(from_unixtime(unix_timestamp(t.ds,'yyyyMMdd'),'yyyy-MM-dd'),-3),1,7),'-','') = t1.ds_month --交错三个月
--交错三个月
```
